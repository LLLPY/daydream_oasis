```python
#导入扩展包
from urllib.request import urlopen

#设置url
url = 'https://www.baidu.com/'

#发送请求，response为发送请后服务器返回的内容
response = urlopen(url)

#读取内容
contents = response.read()

#打印请求
#print(contents.decode())

#打印状态码
print(response.getcode())

#打印真实url
print(response.geturl())

#打印响应头
print(response.info())
```
