```python
n=int(input())
account_set = set()
for i in range(n):
    str_ = input()
    if len(str_) < 6 or len(str_) > 12:
        print('illegal length')
    else:
        if not str_.isalpha():  # 出现了英文以外的字符
            print('illegal charactor')
        else:  # 合法
            if str_ in account_set:
                print('acount existed')
            else:
                print('registration complete')
                account_set.add(str_)
```
