import heapq

# heapq: heap queue

li = [(232332, 12), (1, 'a'), (3, 3), (2, 1)]
heapq.heapify(li)  # 建堆

for i in range(len(li)):
    # heapq.heappush(li,0) #放一个元素到堆中
    print(heapq.heappop(li))  # 每次都是排除最小的一个元素
