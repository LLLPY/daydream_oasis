from functools import wraps
from time import time
from concurrent.futures import ThreadPoolExecutor


# 计算一个函数的执行耗时
def clocked(func):
    @wraps(func)
    def clock(*args, **kwargs):
        time1 = time()
        res = func(*args, **kwargs)
        time2 = time()
        print(f'{func.__name__}耗时：{time2 - time1}。')
        return res
    return clock


# 并行计算每次函数执行所消耗的时间
def dump_clocked(n):
    def clocked(func):
        @wraps(func)
        def clock(*args, **kwargs):
            # 计算每一次执行函数所消耗的时间
            def each_func_cost():
                time1 = time()
                res = func(*args, **kwargs)
                time2 = time()
                return time2 - time1, res

            # 给每一个线程添加额外的任务，通过这个获取每次函数的返回值
            def done(response):
                cost, res = response.result()
                cost_li.append(cost)
                res_li.append(res)

            th_pool = ThreadPoolExecutor(n)
            cost_li = []
            res_li = []

            for i in range(n):
                future = th_pool.submit(each_func_cost)
                future.add_done_callback(done)
            th_pool.shutdown()

            print(f'共测试了{n}次，平均耗时：{sum(cost_li) / n}')
            return res_li

        return clock

    return clocked
