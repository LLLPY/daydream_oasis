```python
n = int(input())

for i in range(n):
    count = 0
    num_list = sorted([int(i) for i in input().split(' ')])
    # 最小的数字减一后如果大于等于0 其他的数字都进行一次减一操作，计算加一

    while num_list[1] - 1 >= 0:
        for i in range(1, 5):
            num_list[i] -= 1

        count += 1
        num_list = sorted(num_list)
    print(count)
```
