from time import time
from numpy import random
def insert_sort(a):
    n = len(a)
    for j in range(1,n):
        i = j
        while i>0:
            if a[i] < a[i-1]:
                a[i],a[i-1] = a[i-1],a[i]
            else:
                break
            i -= 1
