from typing import Optional

'''给你一个链表的头节点 head ，旋转链表，将链表每个节点向右移动 k 个位置。

 

示例 1：


输入：head = [1,2,3,4,5], k = 2
输出：[4,5,1,2,3]
示例 2：


输入：head = [0,1,2], k = 4
输出：[2,0,1]

来源：力扣（LeetCode）
链接：https://leetcode.cn/problems/rotate-list
著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。'''


# Definition for singly-linked list.
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next


class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:

        if not head or k == 0: return head
        li = []
        # 将所有值都添加到一个列表中
        while head:
            li.append(head.val)
            head = head.next

        li_len = len(li)
        # print(li)
        new_li = [0 for i in range(li_len)]  # 保存移动后的元素
        # 可以发现规律 假设当前位置的索引值为old_pos 后移k位 新的位置第(old_pos+k)%len(li)与旧的位置第old_pos对应
        for i in range(li_len):
            new_pos = (i + k) % li_len
            # print(f'old:{i} new:{new_pos}')
            new_li[new_pos] = li[i]
        # print(new_li)

        head = ListNode()
        root = head
        for i in new_li:
            head.next = ListNode(i)
            head = head.next
        return root.next


class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next


class Solution2:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:

        if not head or k == 0: return head

        n = 0
        root = head
        # 寻找尾结点
        while head.next:
            n += 1
            head = head.next
        head.next = root  # 首位相连
        k = k % n
        j = 0
        # 后移k个位置 获取新的头结点
        while j < k:
            root = root.next
            j += 1

        new_root = root
        new_head = new_root
        m = 0
        # 从第n个位置断开
        while m < n:
            new_root = new_root.next
            m += 1
        new_root.next = None
        return new_head


def create_link(li):
    head = ListNode(li[0])
    root = head
    for num in li[1:]:
        head.next = ListNode(num)
        head = head.next
    return root


if __name__ == '__main__':
