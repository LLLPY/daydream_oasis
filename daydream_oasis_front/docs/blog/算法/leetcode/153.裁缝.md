```python
from itertools import permutations

n, m = map(int, input().split(' '))  # 大布料S长度n 客人需要的布料数量m
S = input()  # 大布料
x_li = list(map(int, input().split(' ')))  # m个小布料的长度
s_li = []
for i in range(m):
    s_li.append(input()[:x_li[i]])

res = 0
for com in set(permutations(s_li, m)):
    print(com)
    if ''.join(com) == S:
        res += 1
print(res)
```
