# 计算一个函数的执行耗时
from functools import wraps
from time import time


def clocked(func):
    @wraps(func)
    def clock(*args, **kwargs):
        time1 = time()
        res = func(*args, **kwargs)
        time2 = time()
        print(f'耗时：{time2 - time1}。')
        return res
    return clock
