'''
给定一个排序数组和一个目标值，在数组中找到目标值，并返回其索引。如果目标值不存在于数组中，返回它将会被按顺序插入的位置。
请必须使用时间复杂度为 O(log n) 的算法。
示例 1:
输入: nums = [1,3,5,6], target = 5
输出: 2
示例 2:
输入: nums = [1,3,5,6], target = 2
输出: 1
示例 3:
输入: nums = [1,3,5,6], target = 7
输出: 4
示例 4:
输入: nums = [1,3,5,6], target = 0
输出: 0
示例 5:
输入: nums = [1], target = 0
输出: 0
提示:
1 <= nums.length <= 104
-104 <= nums[i] <= 104
nums 为无重复元素的升序排列数组
-104 <= target <= 104
通过次数560,073提交次数1,214,834
'''
from typing import List


class Solution:
    __slots__ = ()
    def searchInsert(self, nums: List[int], target: int) -> int:
        flag, index = self.half(nums, target)
        if flag:
            return index  # 找到就直接返回
        else:
            if target < nums[0]: return 0  # 如果小于第一个数，就直接插入到第一个位置
            for i in range(index, len(nums)):
                if nums[i - 1] <= target <= nums[i]:  # 在中间，寻找合适的位置并插入
                    return i

            return len(nums)  # 大于所有的数字，直接插入末尾

    # 二分查找 返回的第一个参数表示是否找到 如果找到，第二个参数就是target的索引值；如果没有找到，就是最后查找的位置
    def half(self, li: [int], target: int):
        le, ri = 0, len(li) - 1
        mid = (le + ri) // 2
        while le <= ri:
            mid = (le + ri) // 2
            if target == li[mid]:
                return True, mid
            elif target > li[mid]:
                le, ri = mid + 1, ri
            else:
                le, ri = le, mid - 1
        else:
            return False, mid


if __name__ == '__main__':
    a = [1]
    target = 1
    print(Solution().searchInsert(a, target))
