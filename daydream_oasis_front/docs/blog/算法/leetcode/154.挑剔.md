from collections import OrderedDict

#排序好的字典
a=OrderedDict()
n,m=map(int,input().split(' '))
x_li=input().split(' ')

for i in range(n-1,-1,-1):
    a[str(i+1)]=i+1
# print(a.keys())
for i in range(m):
    a.move_to_end(x_li[i])
print(' '.join(list(a.keys())[::-1]))
