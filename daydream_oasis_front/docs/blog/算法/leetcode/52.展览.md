n = int(input())
flower_li = list(map(int, input().split(' ')[:n]))


def result(li: list):
    res = li[0]^li[1]
    for i in range(1,len(li)-1):
        res+=li[i]^li[i+1]

    print(li,res)

    return res


res = 0
for k in range(2, n + 1):  # 至少两朵花，最多所有的花
    for i in range(n):
        if n - i >= k:
            cur_flower_li = flower_li[i:i + k]
            print(cur_flower_li)
            cur_res = result(cur_flower_li)
            if cur_res>=res:
                res=cur_res

print(res)
