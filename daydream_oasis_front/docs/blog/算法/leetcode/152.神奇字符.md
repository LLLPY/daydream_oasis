n, m = map(int, input().split(' '))
S = input()
s = input()


# 检验两个字符串是否匹配
def match(str1, str2):
    for s1, s2 in zip(str1, str2):
        if s1 == '*' or s2 == '*':
            continue
        if s1 != s2:
            return False
    else:
        return True


res = 0
for i in range(n - m+1):
    # print(S[i:i+m],s,match(S[i:i + m], s))
    if match(S[i:i + m], s):

        res += 1

print(res)
