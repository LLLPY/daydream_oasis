n = int(input())
customers=input().strip().split(' ')
customers_dict = {i+1:customers[i] for i in range(len(customers))}  # 1 2 3 3 5
q = int(input())
res_list = []
for i in range(q):
    left, right, k = input().split(' ')
    left,right=int(left),int(right)
    res = 0
    for m in range(left,right+1):
        if customers_dict[m] == k:
            res += 1
    res_list.append(res)

for res in res_list:
    print(res)
