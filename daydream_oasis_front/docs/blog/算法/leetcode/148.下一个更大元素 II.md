'''给定一个循环数组 nums （ nums[nums.length - 1] 的下一个元素是 nums[0] ），返回 nums 中每个元素的 下一个更大元素 。

数字 x 的 下一个更大的元素 是按数组遍历顺序，这个数字之后的第一个比它更大的数，这意味着你应该循环地搜索它的下一个更大的数。如果不存在，则输出 -1 。

示例 1:

输入: nums = [1,2,1]
输出: [2,-1,2]
解释: 第一个 1 的下一个更大的数是 2；
数字 2 找不到下一个更大的数；
第二个 1 的下一个最大的数需要循环搜索，结果也是 2。
示例 2:

输入: nums = [1,2,3,4,3]
输出: [2,3,4,-1,4]
提示:

1 <= nums.length <= 104
-109 <= nums[i] <= 109
通过次数157,043提交次数238,529
'''
from typing import List


class Solution:
    __slots__ = ()

    def nextGreaterElements(self, nums: List[int]) -> List[int]:
        max_num = max(nums)
        nums_len = len(nums)
        stack = [(nums[0], 0)]
        tmp_dict = {}
        res_li = [0 for i in range(nums_len)]
        i = 1
        while True:
            cur_num = nums[i % nums_len]
            while stack:
                if stack[-1][0] == max_num:
                    value, index = stack.pop()
                    tmp_dict[index] = -1  # 如果是最大的数，直接添加-1即可
                    continue
                if stack[-1][0] < cur_num:
                    value, index = stack.pop()
                    tmp_dict[index] = cur_num
                else:
                    break
            if len(tmp_dict) == nums_len:
                break
            stack.append((cur_num, i % nums_len))  # 记录数字和他的索引
            i += 1

        for i in range(nums_len):
            res_li[i] = tmp_dict[i]
        return res_li


if __name__ == '__main__':
    nums = [1, 1, 1]
    print(Solution().nextGreaterElements(nums))
