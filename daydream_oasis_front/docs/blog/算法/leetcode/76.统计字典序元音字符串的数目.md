# 动态规划
class Solution:
    def countVowelStrings(self, n: int) -> int:
        cur_li = [1, 1, 1, 1, 1]
        for i in range(n - 1):
            li_0 = cur_li[0]
            li_1 = li_0 + cur_li[1]
            li_2 = li_1 + cur_li[2]
            li_3 = li_2 + cur_li[3]
            li_4 = li_3 + cur_li[4]
            cur_li = [li_0, li_1, li_2, li_3, li_4]
        return sum(cur_li)


if __name__ == '__main__':
    print(Solution().countVowelStrings(1))
