from typing import List

'''给定一个包含非负整数的 m x n 网格 grid ，请找出一条从左上角到右下角的路径，使得路径上的数字总和为最小。

说明：每次只能向下或者向右移动一步。
示例 1：

输入：grid = [[1,3,1],[1,5,1],[4,2,1]]
输出：7
解释：因为路径 1→3→1→1→1 的总和最小。
示例 2：

输入：grid = [[1,2,3],[4,5,6]]
输出：12

提示：

m == grid.length
n == grid[i].length
1 <= m, n <= 200
0 <= grid[i][j] <= 100
通过次数362,710提交次数524,239

来源：力扣（LeetCode）
链接：https://leetcode.cn/problems/minimum-path-sum
著作权归领扣网络所有。商业转载请联系官方授权，非商业转载请注明出处。'''


# 动态规划求解
class Solution:
    def minPathSum(self, grid: List[List[int]]) -> int:

        # 依次遍历每一个点，计算到该点的最小距离
        m = len(grid)
        n = len(grid[0])
        for i in range(m):
            for j in range(n):

                # 第一个点的时候
                if i == 0 and j == 0: continue

                # 获取当前点的左边的点
                cur = grid[i][j]

                # 边界问题 当在最左边的时候，最小消耗，只能是当前值加上边的值
                if i == 0:
                    cur_top = grid[i][j - 1]
                    cur_min_cost = cur + cur_top
                    grid[i][j] = cur_min_cost
                    continue

                # 边界问题 当在最上边的时候，最小消耗，只能是当前值加左边的值
                if j == 0:
                    cur_left = grid[i - 1][j]
                    cur_min_cost = cur + cur_left
                    grid[i][j] = cur_min_cost
                    continue

                cur_top = grid[i][j - 1]
                cur_left = grid[i - 1][j]

                cur_min_cost = min(cur_left + cur, cur_top + cur)
                grid[i][j] = cur_min_cost  # 更新当前的最小消耗

        return grid[m - 1][n - 1]


if __name__ == '__main__':
    grid = [[1, 2, 3], [4, 5, 6]]
    print(Solution().minPathSum(grid))
