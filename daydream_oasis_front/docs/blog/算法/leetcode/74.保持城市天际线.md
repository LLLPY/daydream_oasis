from typing import List


class Solution:
    def maxIncreaseKeepingSkyline(self, grid: List[List[int]]) -> int:

        # 计算每一个建筑物可以增加的最大高度
        r = len(grid)
        c = len(grid[0])
        r_max_li = []
        c_max_li = []

        # 第i行的最大数
        for i in range(r):
            max_i = max(grid[i])
            r_max_li.append(max_i)

        # 第j列的最大数
        for j in range(c):
            max_j = grid[0][j]
            for i in range(r):
                if grid[i][j] > max_j: max_j = grid[i][j]
            c_max_li.append(max_j)

        # 第i行第j列最大的数 计算每一个建筑物可以增加的最大高度
        max_sum = 0
        for i in range(r):
            for j in range(c):
                cur_max = min(r_max_li[i], c_max_li[j])
                max_sum += (cur_max - grid[i][j])

        return max_sum


if __name__ == '__main__':
    grid = [[3, 0, 8, 4], [2, 4, 5, 7], [9, 2, 6, 3], [0, 3, 1, 0]]
    print(Solution().maxIncreaseKeepingSkyline(grid))
