```python
'''实现 pow(x, n) ，即计算 x 的 n 次幂函数（即，xn ）。
示例 1：

输入：x = 2.00000, n = 10
输出：1024.00000
示例 2：

输入：x = 2.10000, n = 3
输出：9.26100
示例 3：

输入：x = 2.00000, n = -2
输出：0.25000
解释：2-2 = 1/22 = 1/4 = 0.25
 

提示：

-100.0 < x < 100.0
-231 <= n <= 231-1
-104 <= xn <= 104
通过次数257,210提交次数680,154
'''


#
# class Solution:
#     def myPow(self, x: float, n: int) -> float:
#         res = 1
#         if n < 0:
#             x = 1 / x
#             n = -n
#         for i in range(n):
#             res *= x
#         return res


# 使用分治法
class Solution:
    def myPow(self, x: float, n: int) -> float:
        if n == 0 or x == 1:
            return 1

        if n < 0:
            return 1 / Solution.mypowhelper(x, -n)
        else:
            return Solution.mypowhelper(x, n)

    @classmethod
    def mypowhelper(cls, x, n):

        if n == 1:  # 递归结束的条件
            return x

        if n % 2 == 0:
            half = Solution.mypowhelper(x, n // 2)
            return half * half
        else:
            half = Solution.mypowhelper(x, n // 2)
            return half * half * x


if __name__ == '__main__':
    x = 2
    n = 100000
    print(Solution().myPow(x, n))
```
