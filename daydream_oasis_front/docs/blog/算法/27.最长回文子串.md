```python
'''
给你一个字符串 s，找到 s 中最长的回文子串。
示例 1：

输入：s = "babad"
输出："bab"
解释："aba" 同样是符合题意的答案。
示例 2：

输入：s = "cbbd"
输出："bb"
示例 3：

输入：s = "a"
输出："a"
示例 4：

输入：s = "ac"
输出："a"

提示：

1 <= s.length <= 1000
s 仅由数字和英文字母（大写和/或小写）组成

'''


# 回文的意思是正着念和倒着念一样，如：上海自来水来自海上

class Solution:
    def longestPalindrome(self, s: str) -> str:

        # 如果输入只有一个字符
        if len(s) < 1 or len(s) > 1000:
            return ''
        elif len(s) <= 2:
            return s
        else:
            result = ''
            for i in range(1, len(s) - 1):
                low, high = i, i

                N = 0
                while low >= 0 and high <= len(s):
                    curStr = s[low:high]
                    result = curStr if self.isPalindrome(curStr) and len(curStr) > len(result) else result
                    if N % 2:
                        low -= 1
                    else:
                        high += 1
                    N += 1
                    low = 0 if low <= 0 else low
            return result

            # break

    # 判断一个字符是否是回文:回文的意思是正着念和倒着念一样，如：上海自来水来自海上
    def isPalindrome(self, s: str):
        # # 长度为奇数的字符串
        # if len(s) % 2:
        #     center = len(s) // 2
        #     if s[:center+1] == s[center:][::-1]:
        #         return True
        #     else:
        #         return False
        # # 长度为偶数
        # else:
        #     half = len(s) // 2
        #     if s[:half] == s[half:][::-1]:
        #         return True
        #     else:
        #         return False
        half = len(s) // 2
        first_half = s[:half + 1] if len(s) % 2 else s[:half]
        last_half = s[half:][::-1]
        return first_half == last_half


if __name__ == '__main__':
    s = 'ac'
    print(Solution().longestPalindrome(s))
```
