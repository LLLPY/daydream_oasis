        # 如果吃了当前苹果，就不能吃自己的左右孩子

        # 假设第n个苹果：它的左孩子：2*n+1 右孩子：2*n+2
        tree_len=len(tree)

        self.dfs([0], [], tree, 1)

    def dfs(self, pos, path, tree, eat_cur):
        if pos[0] >= len(tree):
            print(path, sum(path))

            return

        # 吃当前苹果
        if tree[pos[0]] != '#':
            path.append(tree[pos[0]])
        self.dfs([pos[0] + 3], path, tree, 1)
        self.dfs([pos[0] + 1], path, tree, 0)
