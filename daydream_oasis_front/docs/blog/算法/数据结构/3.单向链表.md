#构建节点类

class Node(object):
    def __init__(self,elem):
        #elem指数据元素
        self.elem = elem
        #指向下一个结点的链接域
        self.next = None

#构建单向链表
class SingleLinkList:
    #初始化方法
    #在没有传入值的时候，头结点为空None
    def __init__(self,node=None):
        #判断node是否为空
        if node != None: #头节点不为空
            headNode = Node(node)
            self.__head = headNode
        else:#头节点为空
            self.__head = node

    #在头部添加元素
    def add(self,item):
        #将传入的值构造成节点
        node = Node(item)
        #将新节点的链接域指向头节点
        node.next = self.__head
        #将新节点设置为头节点
        self.__head = node

    #在尾部追加元素
    def append(self,item):
        # 将传入的值构造成节点
        node = Node(item)
        if self.is_empty():#链表为空的情况
            self.__head = node
            node.next = None
        else:#链表不为空的情况
            #寻找尾结点
            curNode = self.__head
            while curNode.next != None: #当该节点指向的下一个节点为空时，就说明这是最后一个节点
                curNode = curNode.next
            #把链表最后一个节点的链接域指向该节点
            curNode.next = node
            #该节点的链接域指向空
            node.next = None


    #在指定位置添加元素
    def insert(self,pos,item):
        #如果输入的pos<=0,则在头部插入
        if pos <= 0:
            self.add(item)
        #如果输入的pos大于链表的长度，则在尾部追加
        elif pos >= (self.length()-1):
            self.append(item)
        else:#中间插入
            #将输入的数据构造成节点
            node = Node(item)
            #找指定节点的前一个节点
            preNode = self.__head
            count = 0
            while count < (pos-1):
                count += 1
                preNode = preNode.next
            #此时的pre为前一个节点
            #将node的链表域指向pre的后一个节点
            node.next = preNode.next
            #将前一个的链表域指向新增的节点node
            preNode.next = node

    #删除结点
    def remove(self,item):
        #将传入的值构成节点
        node = Node(item)
        #辅助结点
        curNode = self.__head

        preNode = None
        while curNode != None:
            if curNode.elem == item:
                #如果要删除的是头结点
                if preNode == None:
                    self.__head = curNode.next
                    return True
                else:
                    #删除
                    preNode.next = curNode.next
                    return True
            else:
                preNode = curNode
                curNode = curNode.next
        return False

    #查找结点是否存在
    def serach(self,item):
        curNode = self.__head
        while curNode != None:
            if curNode.elem == item:
                return True
            curNode = curNode.next
        return False

    #判断链表是否为空
    def is_empty(self):
        return self.__head == None

    #计算链表的长度
    def length(self):
        count = 0
        #定义一个辅助节点,初始值为__head
        curNode = self.__head
        while curNode != None:
            count += 1
            curNode = curNode.next
        return count

        curNode = self.__head

            print(curNode.elem,end='\t')
            curNode = curNode.next

    #将链表信息封装成一个列表并返回
    def return_data_list(self):
        curNode = self.__head
        data_list = []
        while curNode != None:
            data_list.append(curNode.elem)
            curNode = curNode.next
        return data_list
