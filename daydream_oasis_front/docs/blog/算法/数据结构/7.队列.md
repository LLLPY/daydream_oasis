class Queue:

    def __init__(self):
        self.__list = []

    #进队
    def enqueue(self,item):
        self.__list.insert(0,item)

    #出队
    def dequeue(self,item):
        return self.__list.pop()

    def is_empty(self):
        return self.__list == []

    def size(self):
        return len(self.__list)
