class Stack(object):

    def __init__(self):
        #初始化一个私有的空列表
        self.__list = []

    #压栈(向栈中添加元素)
    def push(self,item):
        self.__list.append(item)

    #弹出元素
    def pop(self):
        #返回列表末尾的数据
        return self.__list.pop()

    #返回栈顶元素(即最后添加的一个元素)
    def peek(self):
        return self.__list[-1]

    #判断是否为空
    def is_empty(self):
        return self.__list == []

    #计算栈的大小
    def size(self):
        return len(self.__list)

if __name__ == '__main__':
    stack = Stack()
    print('进栈:')
    stack.push(1)
    print(1)
    stack.push(2)
    print(2)
    stack.push(3)
    print(3)
    print('出栈:')
    print(stack.pop())
    print(stack.pop())
    print(stack.pop())
