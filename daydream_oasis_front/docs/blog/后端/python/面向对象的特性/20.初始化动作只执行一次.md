```python
#让初始化动作只执行一次
#创建一个类对象
class MusicPlayer(object):

    instance = None
    #设置一个类属性，记录初始化动作的执行情况
    init_flag = False

    #初始化设置
    def __init__(self):

        #判断是否执行过初始化动作
        if MusicPlayer.init_flag:
            return

        #如果没有执行就执行
        print("播放器初始化.......")

        #改变init_flag的值
        MusicPlayer.init_flag = True

    #__new__是python内置的静态方法
    def __new__(cls, *args, **kwargs):
        print("创建对象，分配空间")

        #为对象分配空间
        if cls.instance == None:
            cls.instance = super().__new__(cls)

        #返回对象引用
        return cls.instance


#创建多个实例
player1=MusicPlayer()
print(player1)

player2=MusicPlayer()
print(player2)

player3=MusicPlayer()
print(player3)
```
