```python
class Game(object):

    #历史最高分
    top_score = 0

    #定义一个实例属性
    def __init__(self,player_name):
        self.player_name = player_name
    #定义一个静态方法
    @staticmethod
    def show_help():
        print("让僵尸进入大门....")

    @classmethod
    def show_top_score(cls):
        print("历史记录：  %d"%cls.top_score)

    #定义一个实例方法，开始游戏
    def start_game(self):
        print("%s 开始游戏啦...."%self.player_name)

#1查看游戏帮助
Game.show_help()

#2.查看历史成绩
Game.show_top_score()


#3.开始游戏
xiaoming = Game("小明")
xiaoming.start_game()
```
