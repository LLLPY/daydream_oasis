#需求：定义一个工具类---知道创建了多少个工具对象

class Tools(object):

    #利用赋值语句，定义类属性，记录工具对象的总数
    count = 0

    def __init__(self,name):

        self.name = name
        #针对类属性做一个计数
        Tools.count += 1

#创建工具对象
while True:
    tool1 = Tools("斧子")

    print(Tools.count)
