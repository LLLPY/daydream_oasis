```python
class Tools(object):

    #利用赋值语句，定义类属性，记录工具对象的总数
    count = 0

    #定义类方法，查看工具类的数量
    @classmethod
    def show_all_tool(cls):
        print("工具类的总数----> %d"%cls.count)


    def __init__(self,name):

        self.name = name
        #针对类属性做一个计数
        Tools.count += 1

#创建实例对象
tool1 = Tools("斧子")
tool2 = Tools("水桶")


#调用类方法，查看工具类的总数
tool1.show_all_tool()
Tools.show_all_tool()
```
