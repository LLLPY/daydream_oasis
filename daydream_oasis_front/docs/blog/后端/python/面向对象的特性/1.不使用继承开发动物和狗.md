```python
class Animal:

    def eat(self):

        print("吃")

    def drink(self):

        print("喝")

    def run(self):

        print("跑")

    def sleep(self):

        print("睡")


class Dog:

    def eat(self):
        print("吃")

    def drink(self):
        print("喝")

    def run(self):
        print("跑")

    def sleep(self):
        print("睡")

    def bark(self):

        print("汪汪")


#当定义的两个类有很多相同的方法时，如果直接复制粘贴会很麻烦

ahuang = Dog()
ahuang.eat()
ahuang.drink()
ahuang.run()
ahuang.sleep()
ahuang.bark()
```
