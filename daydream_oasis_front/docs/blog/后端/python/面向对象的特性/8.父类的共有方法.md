```python
#子类不能直接调用父类的私有属性
#但在父类内部，使用自己的私有属性和调用私有方法是可以的
#所以可通过在父类内部的公共方法中使用私有属性和调用私有方法
#然后再在子类内部调用包含父类的私有属性和方法的方法
#这样就间接性的使用了父类的私有属性和调用私有方法
class A:

    def __init__(self):

        self.num1 = 100
        self.__num2 = 200

    def __act(self):

        print("这是父类的私有方法")

    def open(self):

        print("这是父类的私有属性num2：%d"%self.__num2)
        self.__act()

class B(A):
    pass

demo = B()
demo.open()
```
