import time


# 装饰器， 计算时间
def timer(fn):
    def wrapper(*args, **kw):
        time_start = time.time()
        result = fn(*args, **kw)
        print(f"Time consuming: {time.time()-time_start}s")
        return result

    return wrapper


# 暴力递归法
@timer
def fib_normal(n):
    def fib_inner(n):
        if n == 1 or n == 2: return 1
        return fib_inner(n - 1) + fib_inner(n - 2)

    return fib_inner(n)


# 备忘录法
@timer
def fib_memo(n):
    memo = [None for _ in range(n + 1)]

    def fib_inner(n):
        if n == 1 or n == 2: return 1
        if memo[n]:
            return memo[n]
        memo[n] = fib_inner(n - 1) + fib_inner(n - 2)
        return memo[n]

    return fib_inner(n)


# dp表格法
@timer
def fib_dp(n):
    dp = [None for _ in range(n + 1)]
    dp[1] = dp[2] = 1

    for i in range(3, n + 1):
        if i <= n:
            dp[i] = dp[i - 1] + dp[i - 2]
        else:
            break

    return dp[n]


# dp表格内存优化版优化
@timer
def fib_dp_advance(n):
    prev, cuur = 1, 1
    for i in range(3, n + 1):
        if i <= n:
            prev, cuur = cuur, prev + cuur
        else:
            break
    return cuur


if __name__ == "__main__":
    # 主程序
    n = int(input("Please enter an integer:"))

    #print(f"Dp method result: {fib_dp(n)}")
    print(f"Dpadvance method result: {fib_dp_advance(n)}")
