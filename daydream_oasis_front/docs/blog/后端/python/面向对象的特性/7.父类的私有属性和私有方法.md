```python
class A:

    def __init__(self):
        self.num1 = 100
        self.__num2 = 200

    def fire(self):
        print("冲啊！！！")

    def __run(self):
        print("跑")

class B(A):

    def act(self):

        #访问父类的非私有属性
        print(self.num1)

        #1.访问父类的私有属性
        print(self._A__num2)

        #2.调用父类的非私有方法
        self.fire()

        #调用父类的私有属性
        self._A__run()

demo = B()
demo.fire()
demo.act()
#子类对象不能在自己的方法内部，直接访问父类的私有属性或调用私有方法

```
