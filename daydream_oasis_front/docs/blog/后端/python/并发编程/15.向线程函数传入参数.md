```python
import _thread
from time import sleep

def fun1(name,timeout):
    print('start:',name)
    sleep(timeout)
    print('end:',name)

def fun2(name,timeout):
    print('start:',name)
    sleep(timeout)
    print('end:',name)

if __name__ == '__main__':

    #创建线程1
    _thread.start_new_thread(fun1,('线程1',2))
    #创建线程2
    _thread.start_new_thread(fun2,('线程2',3))
    sleep(6)
    print('over!')
```
