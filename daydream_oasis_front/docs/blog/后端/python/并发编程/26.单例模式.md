    instance = None

    def __init__(self):
        pass

    def __new__(cls, *args, **kwargs):
        # 在创建一个新的对象时，首先判断有没有已经存在的对象
        # 如果没有存在的对象就创建一个新的对象
        # 这样就保证了所有的对象使用的都是同一个地址
