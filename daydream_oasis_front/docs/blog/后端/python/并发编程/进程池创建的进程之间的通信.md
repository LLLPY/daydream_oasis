```python
#导入Queue
from multiprocessing import Manager,Pool
from time import sleep

#定义子写入进程
def writer(q):
    q_list = ['a', 'b', 'c', 'd', 'e']
    for i in q_list:
        print('写入:',i)
        q.put(i)
        sleep(1)

#定义读取的进程
def reader(q):
    for i in range(q.qsize()):
        print('读取:',q.get())
        sleep(1)

if __name__ == '__main__':

    # 定义一个队列
    q = Manager().Queue()
    #创建进程池
    p = Pool(2)
    #向进程池中添加任务
    p.apply(writer,(q,))
    p.apply(reader,(q,))

    #关闭进程池
    p.close()
    p.join()
    print('over！')
```
