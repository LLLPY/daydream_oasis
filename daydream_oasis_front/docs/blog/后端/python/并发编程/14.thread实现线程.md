```python
#导入_thread模块
import _thread
from time import sleep
#定义线程函数1
def fun1():
    print('start:fun1')
    sleep(2)
    print('end:fun1')


def fun2():
    print('start:fun2')
    sleep(3)
    print("end:fun2")

if __name__ == '__main__':

    #创建线程1
    _thread.start_new_thread(fun1,())
    #创建线程2
    _thread.start_new_thread(fun2,())

    #等待线程的结束，主进程再结束
    sleep(6)
    print('over!')


```
