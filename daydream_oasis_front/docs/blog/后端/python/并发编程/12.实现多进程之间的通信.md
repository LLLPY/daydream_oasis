```python
#导入Queue
from multiprocessing import Queue,Process
from time import sleep
q = Queue(1)
#定义进程1
def test1():
    print('start:test1')
    m = q.get()
    print('test1从队列中get的数值是:', m)
    m += 10
    q.put(m)
    print('队列中的数值为:',m)
    print('end:test1')
    sleep(2)

def test2():
    print('start:test2')
    n = q.get()
    print('test2从队列中get的数值是:',n)
    n += 10
    q.put(n)
    print('队列中的数值为:',n)
    print('end:test2')
    sleep(2)


if __name__ == '__main__':
    # 创建一个队列

    # 传入数据到队列中
    q.put(10)
    #创建进程1
    p1 = Process(target=test1)
    #创建进程2
    p2 = Process(target=test2)
    #启动进程
    p1.start()
    p1.join()
    p2.start()
    #等待子进程结束
    p2.join()
    #主进程
    print('子进程结束后,队列中的值为:',q.get())
    print('over！')
```
