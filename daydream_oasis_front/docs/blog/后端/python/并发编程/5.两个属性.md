from multiprocessing import *
from datetime import *
from time import sleep
def clock(ti):

    for i in range(3):
        print('当前时间是:',datetime.now().strftime("%m/%d/%Y %H:%M:%S"))
        sleep(ti)


if __name__ == '__main__':

    ti = int(input('时间间隔:'))
    p = Process(target=clock,args=(ti,))
    p.start()
    p.join()
    print('p.name:',p.name)
    print('p.pid:',p.pid)
