from time import sleep

def sing():
    for i in range(3):
        print('正在唱歌...%d'%i)
        dance()
        sleep(1) #唱歌一次休息一秒钟

def dance():
    for i in range(3):
        print('正在跳舞...%d'%i)
        sleep(1) #跳舞一次休息一秒钟

if __name__ == '__main__':
    sing()
    dance()
#从运行结果可以看到：跳舞和唱歌是分开进行的，不是同时进行的
