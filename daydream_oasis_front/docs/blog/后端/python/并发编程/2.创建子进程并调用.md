```python
#导入Process模块
from multiprocessing import Process

#定义子进程的方法
def little_procese():
    print('子进程正在运行....')



if __name__ == '__main__':
    print('主进程正在运行.....')

    p = Process(target=little_procese())#创建子进程实例对象
    p.start() #让子进程调用start方法进行运行
```
