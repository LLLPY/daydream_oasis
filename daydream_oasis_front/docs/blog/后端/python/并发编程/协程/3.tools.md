from functools import wraps
from time import time


# 计算一个函数的执行时间
def clocked(func):
    @wraps(func)
    def clock(*args, **kwargs):
        time1 = time()
        res = func(*args, **kwargs)
        time2 = time()
        print(f'exec {func.__name__} cost {time2 - time1}.')
        return res

    return clock
