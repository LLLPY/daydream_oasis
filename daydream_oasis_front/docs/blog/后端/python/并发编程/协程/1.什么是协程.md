#与线程相比，协程是更小的执行单元，具有更小的开销。
#与协程相关的库就是asyncio，asyncio 往往是构建 IO 密集型和高层级 结构化 网络代码的最佳选择。


#demo
import asyncio

async def main():
    print('hello!')
    await asyncio.sleep(1)
    print('world!')



if __name__ == '__main__':
    asyncio.run(main())
