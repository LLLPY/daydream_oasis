import asyncio

'''
asyncio.shield(aw)
#保护一个 可等待对象 防止其被 取消。
'''


async def say_hello(word):
    print(f'hello,{word}!')

async def main():
    task = asyncio.create_task(say_hello('Python'))
    await asyncio.shield(task)  # 任务不会被取消
    task.cancel()  # 取消任务


asyncio.run(main())
