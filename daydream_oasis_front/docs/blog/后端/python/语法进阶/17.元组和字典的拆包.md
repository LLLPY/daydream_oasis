def demo(*args,**kwargs):

    print(args)
    print(kwargs)

gl_tuple = (1,2,3,4,5)
gl_dict = {"name":"小明","age":18}
demo(*gl_tuple,**gl_dict)

"""
元组和字典的拆包
.在调用带有多值参数的函数时，如果希望：
   。将一个元组变量，直接传递给args
   。将一个字典变量，直接传递给kwargs
.就可以使用拆包，简化参数的传递，拆包的方式：
    。在元组变量前加一个*
    。在字典变量前加两个**


"""
