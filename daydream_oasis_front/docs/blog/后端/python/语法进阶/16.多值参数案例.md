```python
#需求：
     #定义一个函数sum_numbers，可以接受任意多个整数
     #功能要求：将传递的所有数字累加并且返回结果

def sum_numbers(*list_num):
    sum = 0
    # print(list_num[0])
    # print(list_num[1])
    # print(list_num)
    #循环遍历进行累加
    for number in list_num:
        sum += number
    return  sum

print("结果为%d"%sum_numbers(1,2,12,23,34,45))
```
