```python
def demo(num,list_num):

    print("函数开始")
    #num = num + num
    num += num

    #  += 对列表进行操作时，不会做相加再赋值的操作
    #  相当于方法extend,不会修改变量的引用
    list_num += list_num
    print(num)
    print(list_num)
    print("函数完成")

list_number = [12,23,34]
print(list_number)

demo(6,list_number)
print(list_number)
```
