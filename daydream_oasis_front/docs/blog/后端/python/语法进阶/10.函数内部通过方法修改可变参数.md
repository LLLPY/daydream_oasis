```python
#如果传递的参数是可变类型，在函数内部，使用方法修改了数据内容，
#同样会影响到外部数据

def demo(list_num):

    list_name = ["小明","小美","小张"]

    list_num.extend(list_name)
    list_num.append(9)
    print("在函数内部")
    print(list_num)

list_number = [1,2,3]
demo(list_number)

print("在函数外部")
print(list_number)
demo(list_number)
```
