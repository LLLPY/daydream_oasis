```python
"""
题目要求：
 1.有两个整数变量 a=6，b=100
 2.不使用其他变量，交换两个变量的值
"""

a = 6
b= 100

#解法1
a = a + b
b = a - b
a = a - b
print(a)
print(b)

#解法2
#利用元组，python专有
(a,b) = (b,a)
print(a)
print(b)
```
