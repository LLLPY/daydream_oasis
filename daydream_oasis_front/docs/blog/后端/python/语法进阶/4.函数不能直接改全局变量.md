```python
#全局变量
num = 10
def print_num():
    #希望修改全局变量
    #python中不能直接修改全局变量的值
    #如果使用赋值语句，会在函数内部，定义一个局部变量

    #如果希望在函数颞部改变全局变量的值，使用global声明一下变量即可
    #global关键字会告诉后面的变量是一个全局变量
    #再使用赋值语句时，就不会创建局部变量
    global num
    num = 12
    print("num--->%d"%num)

print("全局变量的值为：%d"%num)
print_num()

```
