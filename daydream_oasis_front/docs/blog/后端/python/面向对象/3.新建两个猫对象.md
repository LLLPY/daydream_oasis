```python
class Cat():
    def eat(self):
        print("我爱吃鱼")
    def drink(self):
        print("我还需要喝水")

lazy_cat = Cat()
lazy_cat.drink()
lazy_cat.eat()
#定义属性时
    #可以使用.属性名，利用赋值语句就可以了
lazy_cat.name = "大懒猫"

tom = Cat()
tom.eat()
tom.drink()
tom.name = "Tom"

print(lazy_cat)
print(tom)
#两只猫的地址不一样，说明是两只不同的猫
```
