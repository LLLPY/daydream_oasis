```python
#定义家具类
class Houseitem:

    #初始化设置
    def __init__(self,name,area):
        self.name = name
        self.area = area

    #返回各个家具的有关信息
    def __str__(self):
        return "【%s】占地 %.2f 平米"%(self.name,self.area)

bed = Houseitem("席梦思",4.0)
chest = Houseitem("衣柜",2.0)
table = Houseitem("餐桌",1.5)
print(bed)
print(chest)
print(table)
```
