```python
class Women:

    def __init__(self,name):

        self.name = name

        self.__age = 20

    def secret(self):
        #在对象内部，是可以访问对象的私有属性的
        print("%s的年龄是%d"%(self.name,self.__age))

#私有属性，在外界不能被直接访问
xiaomei = Women("小美")
# print(xiaomei.__age)
#私有方法同样不能被外界直接访问
xiaomei.secret()
```
