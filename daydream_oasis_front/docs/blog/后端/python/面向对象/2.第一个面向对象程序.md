```python

"""
    #需求：小猫爱吃鱼，小猫要喝水
    分析：
        1.定义一个猫类
        2.定义两个方法eat和drink
        3.按照需求——不需要定义属性
"""
#定义猫类
class Cat():
    def eat(self):
        print("我爱吃鱼")
    def drink(self):
        print("我还需要喝水")

#创建对象
tom = Cat()
tom.eat()
tom.drink()
#十六进制
print(tom)

#十进制
addr = id(tom)
print("%d"%addr)
```
