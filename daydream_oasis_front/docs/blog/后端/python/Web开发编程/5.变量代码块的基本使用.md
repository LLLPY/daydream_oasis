```python
from flask import Flask,render_template

app=Flask(__name__)

@app.route('/',methods={'GET','POST','PUT'})
def hello():

    #传入一个网址
    url = "www.baidu.com"
    #传入一个列表
    my_list = [2,4,7,"i love you "]
    #传入一个字典
    my_dict = {"name":'百度',
               'url':'www.baidu.com'}
    return render_template("index.html",url=url,my_list=my_list,my_dict=my_dict)

if __name__ == '__main__':
    app.run(debug=True)
```
