```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>动画</title>
    <script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>

    <script>

        /*
        *
        * animate({params},speed,callback);
        * 必须的params参数定义形成动画的css属性
        * 可选的speed参数定义效果的时长
        * 可选的callback参数定义回调函数
        *
        *
        *
        * */

        $(function () {
            $("button").click(function () {
                let box = $("#box");
                box.animate(
                    {
                        /* "left": "200px",
                         "width": "150px",
                         "height": "150px",
                         "opacity":"0.5",*/
                        // "backgroundColor":"red", 如果需要生成颜色动画，您需要从 jquery.com 下载 颜色动画 插件。

                        /*使用相对值*/
                        /*"left": "+=200px",
                        "width": "+=150px",
                        "height": "+=150px",
                        "opacity":"-=0.5",*/

                        /*使用预定义的值  把属性的动画值设置为show,hide,toggle*/
                        /*"width":"toggle",
                        "height":"hide",*/


                    },);
                /*使用队列 默认地,jQuery提供针对动画的队列功能,这意味着如果您在彼此之后编写多个animate()调用
                     * jQuery会创建包含这些方法调用的"内部"队列,然后逐一运行这些animate调用
                     *
                     * */

                box.animate({"left": "+=400px", "width": "+=50px", "height": "+=50px"}, 800);
                box.animate({"top": "+=400px", "width": "+=50px", "height": "+=50px"}, 700);
                box.animate({"left": "-=400px", "width": "-=50px", "height": "-=50px"}, 600);
                box.animate({"top": "-=400px", "width": "-=50px", "height": "-=50px"}, 500, function () {
                    // alert("回到原点");
                });


            });


        });


    </script>

    <style>
        button {
            width: 80px;
            height: 40px;
            text-align: center;
            line-height: 40px;
        }

        #box {
            position: absolute;
            top: 100px;
            left: 100px;
            width: 100px;
            height: 100px;
            background-color: green;
        }


    </style>


</head>
<body>

<button>开始运动</button>

<div id="box"></div>


</body>
</html>
```
