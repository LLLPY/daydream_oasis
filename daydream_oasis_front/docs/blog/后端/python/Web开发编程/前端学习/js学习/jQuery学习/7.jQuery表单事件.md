```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>jQuery表单事件</title>
    <script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>

    <script>
        /*
        *
        * 表单事件
        *
        * 1.submit 当表单提交时会触发该事件
        * 2.change(仅适用于比表单字段) 当元素的值发生改变时触发该事件
        *   当用于select元素时,change事件会在选择某个选项时发生,当用于text field或text area时,change事件
        *   会在元素失去焦点时发生
        *3.focus 当元素获得焦点时(当通过鼠标选中元素或者通过tab键定位到元素时),发生focus事件
        *4.blur 当元素失去焦点时发生该事件
        *
        *
        *
        * */


        /*文档预加载*/
        $(function () {

            form_tag = $('form');
            accout_tag = $('input:first-child');
            pwd_tag = $('input:last-child');
            /*提交表单时,提示时候确认提交*/
            form_tag.submit(
                function () {
                    alert('是否确认提交');
                }
            );


            /*鼠标移出输入框提示*/
            accout_tag.change(
                function () {
                    alert('账号输入框内容发生改变');

                }
            );
            /*鼠标聚焦密码框时提示*/
            pwd_tag.focus(
                function () {
                    alert('已聚焦密码框');
                }
            );

            /*账号框失去焦点*/
            accout_tag.blur(
                function () {
                alert('账号输入框失去焦点');
                }


            );


        });


    </script>


</head>
<body>

<form action="#">

    账号: <input type="text">
    密码: <input type="password">

    <button>登录</button>
</form>


</body>
</html>
```
