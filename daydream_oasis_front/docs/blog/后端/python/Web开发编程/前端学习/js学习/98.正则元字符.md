```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>正则-元字符</title>
</head>
<script>

    window.onload = function () {

        //元字符:在正则表达式中有特殊含义的字符
        //单个字符和数字的元字符
        /*
        * . 匹配任意单个字符
        * [范围] 匹配范围内的任意单个字符
        * [0-9] 匹配单个数字
        * [a-zA-Z0-9_] 匹配数字字母下划线
        * [^范围] 匹配范围外的任意一个字符
        * [^0-9] 匹配任意一个非数字字符
        * \w 匹配单个的数字字母下划线 等价于[a-zA-Z0-9_]
        * \W 匹配非数字字母下划线的单个字符
        * \d 匹配单个数字
        * \D 匹配单个非数字字符
        *
        * */

        //重复字符 x(任意的单个字符)
        /*
        * x? 匹配0个或者一个x
        * x+ 匹配至少一个x
        * x* 匹配任意个x字符
        * x{m,n} 匹配至少m个至多n个x字符
        * x{n} 匹配n个x字符
        * (xyz) 将xyz看成一个整体,当做一个字符处理
        *
        * 锚字符
        * ^ 行首匹配
        * ^x 必须以x开头
        * $ 行尾匹配
        * x$ 必须以x结尾
        *
        *
        * */

        //空白字符
        /*
        *
        * /s 匹配任意一个空白字符
        * /S 匹配任意一个非空白字符
        *
        * */

        //修饰符
        /*
        * i 忽略大小写
        * g 全局匹配
        * m 换行匹配
        *
        *
        * */


        var strs='1.google\n2.b23ing\n3.bai123du';
        //利用m修饰符和行首匹配用*替换掉每一行的第一个数字,而中间的数字不变
        var demo=new RegExp('^\\d','gm');
        var demo2=/^\d/gm;
        alert(demo); // /^\d/gm /^\d/gm
        alert(demo2);
        alert(demo===demo2);
        alert(strs.replace(demo,'*'));
        alert(strs.replace(demo2,'*'));


    }


</script>


<body>

</body>
</html>
```
