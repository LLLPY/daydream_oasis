```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>条件渲染</title>
    <script src="js/vue.js"></script>
</head>
<body>

<div id="root">
    当前n={{n}}
    <button @click="n++">点我n++</button>

    <div v-if="n===1">python(n=1显示)</div>
    <div v-else-if="n===2">golang(n=2显示)</div>
    <div v-else-if="n===3">vue(n=3显示)</div>
    <div v-else>hello world!</div>

    <div v-show="n>=1">哈哈哈哈</div>

    <!--对于如下内容可以使用if配合template使用来简化,template标签不会影响页面的结构-->
    <h3 v-if="n===4">html</h3>
    <h3 v-if="n===4">css</h3>
    <h3 v-if="n===4">javascript</h3>

    <template v-if="n===4">
        <h3>html</h3>
        <h3>css</h3>
        <h3>javascript</h3>
    </template>

</div>

<script>

    /*
    *
    * 条件渲染
    *   1.v-if
    *       写法：
    *           （1）.v-if="表达式"
    *           （2）.v-else-if="表达式"
    *           （3）.v-else="表达式"
    *       适用于：切换频率较低的场景。
    *       特点：不展示的DOM元素直接被移除。
    *       tips：v-if可以和v-else-if,v-else一起使用，但要求结构不能被“打断”
    *
    * 2.v-show
    *       写法：v-show="表达式"
    *       适用于：切换频率较高的场景。
    *       特点：不展示的DOM元素未被移除，仅仅是使用样式隐藏掉
    *
    *
    *
    * */


    new Vue({
        el: '#root',
        data: {
            n: 0,
        }
    })

</script>


</body>
</html>
```
