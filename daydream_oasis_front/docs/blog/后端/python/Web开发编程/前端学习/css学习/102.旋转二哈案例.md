```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>旋转二哈案例</title>

    <style>

    body {
             perspective:1000px;   /*透视的属性值给父级元素 值越小越明显*/

    }

        section {
            width:350px;
            height:250px;
            margin:100px auto;
            background:url(images/erha.jpg) no-repeat;
            background-size:cover;
            position:relative;
            transform-style:preserve-3d; /*3d显示效果 默认是2d(让父盒子里的子盒子以3d效果显示)*/
            transition:all 6s linear;
        }

        section:hover {
            transform:rotateY(360deg);

        }

        div {
            width:350px;
            height:250px;
            background:url(images/erha.gif) no-repeat;
            background-size:cover;
            position:absolute;

        }

        div:nth-child(1){
            transform:rotateY(0deg) translateZ(400px);  /*让每个div盒子到section的z轴距离都是400px*/
        }
         div:nth-child(2){
            transform:rotateY(60deg) translateZ(400px);
        }
         div:nth-child(3){
            transform:rotateY(120deg) translateZ(400px);
        }
         div:nth-child(4){
            transform:rotateY(180deg) translateZ(400px);
        }
         div:nth-child(5){
            transform:rotateY(240deg) translateZ(400px);
        }
         div:nth-child(6){
            transform:rotateY(300deg) translateZ(400px);
        }


    </style>

</head>
<body>

<section>
    <div></div>
    <div></div>
    <div></div>
    <div></div>
    <div></div>
    <div></div>

</section>
</body>
</html>
```
