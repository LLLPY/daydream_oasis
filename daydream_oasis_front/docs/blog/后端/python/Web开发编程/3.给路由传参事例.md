```python
#1.导入Flask模块
from flask import Flask

#2.创建一个Flask实例对象
#__name__是为了确定资源所在的路径
app = Flask(__name__)

#3.定义路由和视图函数
#路由定义是通过修饰器实现的
@app.route('/orders/<order_id>')
def demo(order_id):
    return "order_id %s" % order_id

#4.启动程序
if __name__ == '__main__':
    app.run()

```
