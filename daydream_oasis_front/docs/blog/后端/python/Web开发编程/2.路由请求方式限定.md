```python
# 1.导入Flask模块
from flask import Flask

# 2.创建Flask实例对象
# 序传入__name__，作用是为了确定资源所在的路径
demo = Flask(__name__)


# 3.定义路由及视图函数
# Flask中定义路由是通过修饰器实现的
# 路由默认只支持GET，如果需要增加，需要自行指定
@demo.route('/', methods=['GET', 'POST', 'PUT'])
def hello():
    return "hello flask"


# 4.启动程序
demo.run()

```
