```python
import os
import re

f = os.popen('tasklist/nh','r')
pattern = r'([\w.]+([\w.]+)*)\s\s+(\d+) \w+\s\s+\d+\s\s+([\d+,]+ K)'

list = []
for eachline in f:
    x = re.findall(pattern,eachline.rstrip())
    if x != []:
        list.append(x)
f.close()
print('当前正在执行的任务:')
for i in list:
    print(i)
```
