```python
import re

#贪婪模式 ：尽可能多的匹配字符(默认情况下是贪婪模式)
#非贪婪模式 ：尽可能少的匹配字符

phone = '023-2332434'

#贪婪模式下的匹配
pattern = r'\d+'
x = re.search(pattern,phone)
print(x.group())

#非贪婪模式下的匹配
pattern = r'\d+?'
x = re.search(pattern,phone)
print(x.group())
```
