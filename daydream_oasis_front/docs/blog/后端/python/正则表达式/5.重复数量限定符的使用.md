```python
import re

#匹配第一个字符是大写字母，后面的可以是0个或多个小写字母
s = 'AafDfjai'
pattern = '[A-Z][a-z]*'
x = re.match(pattern,s)
print(x.group())

#匹配有效的变量名(可以由字母,数字,下划线组成，且不能以数字开头)

pattern = '[a-zA-Z_][a-zA-Z0-9_]*'
pattern = '[a-zA-Z_]\w*'
s1 = '12ewad'
s2 = '_12ewad'
s3 = 'a_12ewad'
x1 = re.match(pattern,s1)
x2 = re.match(pattern,s2)
x3 = re.match(pattern,s3)
if x1 != None:
    print(x1.group())
if x2 != None:
    print(x2.group())
if x3 != None:
    print(x3.group())

#匹配1-99的数字
pattern = '[1-9][0-9].?\d'
s1 = '121'
s2 = '92'
s3 = '12.8'
x1 = re.match(pattern,s3)
print(x1.group())
```
