```python
import re

s = 'adadfeafjhaj'
pattern = r'ad|ea|ha' # |在这里是逻辑或ed意思，表示既可以匹配ad，也可以匹配ea，也可以匹配ha
x = re.search(pattern,s)
print(x)

#匹配0-100的数字 0-99|100

pattern = r'[1-9]?\d|100'
s = '100'
x = re.search(pattern,s)
print(x)
```
