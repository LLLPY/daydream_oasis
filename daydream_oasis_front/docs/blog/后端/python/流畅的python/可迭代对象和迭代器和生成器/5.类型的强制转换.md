```python
# 使用type进行类型的强制装换
a = 'hello world'  # 字符串类型
b = 10  # 整数类型

print(f'转之前a的类型:{type(a)},b的类型：{type(b)}')

# 将b转成a的类型
print(type(a)(b))
print(type(type(a)(b)))

```
