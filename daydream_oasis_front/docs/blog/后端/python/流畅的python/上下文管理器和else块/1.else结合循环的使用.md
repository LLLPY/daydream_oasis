```python
# 循环只有正常结束才会执行else
for i in range(10):
    print(i)
else:
    print('循环正常结束！')

# 如果循环被break打断,else就不会被执行
# for i in range(10):
#     print(i)
#     break
# else:
#     print('反正写了也不会执行...')

i = 0
while True:
    print(i)
    i += 1
    if i > 2:
        break
else:
    print('else执行了！')

```
