# 让类的实例成为可调用的

class Cat:
    instance = None

    def __init__(self, name):
        self.name = name

    def __new__(cls, *args, **kwargs):
        if not cls.instance:
            cls.instance = object.__new__(cls)

        return cls.instance

    def __call__(self, *args, **kwargs):
        print(f'hello,my name is {self.name}!')


if __name__ == '__main__':
    Tom = Cat('Tom')
    print(callable(Tom))
    Tom()
