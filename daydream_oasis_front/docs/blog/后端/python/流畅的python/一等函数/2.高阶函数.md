# reduce函数

# sum和reduce的通用思想是把某个操作***连续应用***到序列的元素上，累计之前的结果，把一系列的值规约成一个值。

from operator import add
from functools import reduce

print(reduce(add, range(100)))
print(sum(range(100)))
