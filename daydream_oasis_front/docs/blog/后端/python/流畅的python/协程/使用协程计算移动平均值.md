def averager():
    total = 0
    count = 0
    average = 0
    while True:
        term = yield average  # send()的值会传给term
        total += term
        count += 1
        average = total / count
        print(average)


if __name__ == '__main__':
    my_avg = averager()
    print(next(my_avg))  # 激活协程 可以看到协程产出的初始值是average的初始值为None

    cur_avg = my_avg.send(1)
    cur_avg = my_avg.send(2)
