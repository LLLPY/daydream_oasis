# python中数据的属性和处理数据的方法统称为属性
class Cat:

    def __init__(self, name):
        self.name = name

    def run(self):
        print('小碎步...')


if __name__ == '__main__':
    tom = Cat('Tom')
    del tom.name
    # print(tom.__getattribute__('name'))  # 对象的属性
    print(tom.__getattribute__('run'))  # 对象的方法

    tom.run=None
    print(tom.__getattribute__('run'))  # 对象的方法
    del tom.run
    print(tom.__getattribute__('run'))  # 对象的方法
    tom.run()
