```python
from functools import reduce
from operator import xor

a = [i for i in range(10)]

# 1.使用for循环
res = 0
for i in a:
    res ^= i
print(res)

# 2.lambda结合reduce
res = reduce(lambda a, b: a ^ b, a)
print(res)

# 3.异或运算符结合reduce
res = reduce(xor, a)
print(res)

```
