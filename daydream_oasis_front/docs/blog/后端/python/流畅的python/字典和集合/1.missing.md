class MyDict(dict):

    # 当调用a[k]方法取值时，如果k不存在会调用这个方法
    def __missing__(self, key):
        print(f'字典中没有{key}')

class Myset(set):

    def __missing__(self, key):
        print(f'集合中没有{key}')

class Cat:

    def __hash__(self):
        return id(self)

    def __eq__(self, other):
        return id(self) == id(other)


if __name__ == '__main__':
    mydict = MyDict([('a', 1)])
    mydict.get()
    print(mydict[2])
    print(type(mydict.keys()))
    a = Cat()
    print(hash(a), id(a))
