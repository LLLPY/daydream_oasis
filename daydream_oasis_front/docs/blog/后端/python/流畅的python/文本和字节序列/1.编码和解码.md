```python
# 编码：将码位转换成字节序列的过程
# 解码：把字节序列变成码位的过程

s = 'apple'

s = s.encode('utf8')
print(s, type(s))
s = s.decode('utf8')
print(s, type(s))

s_byte=bytes(s,encoding='utf8')
print(s_byte,type(s_byte))

s_bytearray=bytearray(s,encoding='utf8')
print(s_bytearray,type(s_bytearray))
```
