```python
import struct

fmt = '<3s3sHH'

with open('demo.jpg', 'rb') as f:
    img = memoryview(f.read())
    header = img[:10]
    print(bytes(header))
    info_tuple=struct.unpack(fmt, header)
    print(info_tuple)
    print(info_tuple[0].decode('utf8',errors='replace    '))

```
