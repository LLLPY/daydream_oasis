```python
from time import time

a = [i for i in range(3*10**7)]
b = tuple(a)
print(f'增量操作前a的id={id(a)},b的id={id(b)}')

start=time()
a *= 2
mid=time()
b *= 2
end=time()
print(f'增量操作后a的id={id(a)},b的id={id(b)}')
print(f'列表增量操作耗时：{mid-start}')
print(f'元组增量操作耗时：{end-mid}')
print(f'列表增量的速度时元组增量的{(end-mid)/(mid-start)}')
```
