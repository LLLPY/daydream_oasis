```python
# 根据flag来决定是否打印函数名
def print_func_name(flag=False):  # 装饰器加工厂

    def decorate(func):  # 真正的装饰器
        if flag:
            print(f'因为flag=True，所以打印了{func.__name__}')
        return func

    return decorate


@print_func_name(flag=True)
def hello(word):
    print(f'hello {word}')


@print_func_name()
def say(word):
    print(f'say {word}')


if __name__ == '__main__':
    hello('python')
    say('hello')

```
