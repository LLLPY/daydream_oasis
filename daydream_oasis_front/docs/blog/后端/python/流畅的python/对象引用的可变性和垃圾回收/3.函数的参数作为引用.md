def _(x):
    print(id(x))
    x += x
    return x


if __name__ == '__main__':
    a = [1, 2, 3]  # 得到  [1, 2, 3,1, 2, 3]
    a = (1, 2, 3)  # 得到(1,2,3)
    a = 1  # 得到 1
    print(id(a))
    _(a)
    print(a)

    '''
    个人理解：
        从结果可以看到，函数中的参数是是对传入的变量的引用，音因此，两者的id是相同的
    '''
