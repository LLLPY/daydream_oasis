```python
from collections.abc import Sequence

a = [1, 2, 3]
# isinstance 检测a是不是序列
print(isinstance(a, Sequence))

# issubclass 检测某个类是不是另一个类的子类，从而确定其类型
print(issubclass(list, Sequence))

# type
print(type(a) == list)

```
