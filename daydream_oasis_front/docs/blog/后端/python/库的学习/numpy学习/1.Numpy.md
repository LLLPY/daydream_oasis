```python
#导入numpy模块
import numpy as np #给他去别名为np
import math

#创建一个ndarry对象
a = np.arange(10) #a是一个一维数组（0到9的整数），arange左闭右开
print(type(a))
print(a)

b = []
#给数组里的每一个数开平方
#循环遍历每一个数进行开平方处理
for i in a:
    b.append(math.sqrt(i))
print(b) #调用math里的开平方的方法

#直接使用numpy提供的方法对ndarray对象里的每一个数进行开平方
print(np.sqrt(a))
```
