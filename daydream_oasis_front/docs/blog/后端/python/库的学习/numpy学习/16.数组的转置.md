```python
import numpy as np

#数组的转置：行列互换

a = np.arange(1,13).reshape((3,4))
print('这是起初的二维数组a:\n',a)

print('方法一:\n')
b = a.reshape((4,3))
print('这是转置后的数组a:\n',b)

print('方法二:\n')
c = a.transpose()
print(c)

print('方法三:\n')
d = a.T
print(d)

e = a.reshape((1,4,3)) #(0,1,2)
f = e.transpose()
print(f,f.shape)
g = e.transpose((2,1,0))
print(g)
print(g.shape)
```
