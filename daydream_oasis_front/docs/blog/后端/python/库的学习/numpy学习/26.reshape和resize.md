```python
import numpy as np


a=np.array([[1,2,3],[4,5,6]])


'''
reshape和resize的区别在于,reshape会生成一个数组,而
reszie是在原来数组的基础上进行修改

'''

#将二维数组转换成一维的
# reshape方法 注意:reshape是生成一个新数组
b=a.reshape((6))
print(b)

#resize方法 注意:resize是修改原有的数组 所以返回值为None
c=a.resize((6,))
print(a)




```
