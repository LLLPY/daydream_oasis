```python
import numpy as np

# 接口
'''
ndarrayastype(type)
参数:
    type:需要转换成的类型

'''


a=np.array([[1,0.12,23],[0.222,1,3.2323]])
print(a)

#将a转换成int64型的
b=a.astype(np.int64)
print(b)

#将a转成字符型
c=a.tostring()
print(c)

#转换成bytes类型
d=a.tobytes()
print(d)
```
