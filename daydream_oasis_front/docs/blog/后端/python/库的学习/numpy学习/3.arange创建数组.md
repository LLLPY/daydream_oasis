```python
import numpy as np

#使用arange创建数组 左闭右开
a = np.arange(1,10,2,float) #arange有四个参数 start默认值为0 stop需要传入 dtype默认为用户输入的类型 step默认为1
print(a)

#使用range创建数组

b = list(range(1,10,2)) #range里面有三个参数 start end step默认步长是1 左闭右开
print(range(1,10,2) )

print(b)

```
