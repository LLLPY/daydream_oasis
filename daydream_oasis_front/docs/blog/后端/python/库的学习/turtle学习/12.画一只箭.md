def arrow(pos,pencolor,fillcolor): #位置 笔的颜色 填充色 角度

    width(3)
    setpos(pos[0], pos[1])
    setheading(0)
    down()
    color(pencolor, fillcolor)

    begin_fill()

    length=350 #箭身的长度

    #箭尾
    left(150)
