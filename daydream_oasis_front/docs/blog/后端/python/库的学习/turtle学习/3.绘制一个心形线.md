from turtle import *
from math import sin, cos, pi
from numpy.ma import arange

def drawHeart():
    x_list=[]
    y_list=[]
    for i in arange(0,2*pi+0.1,0.1):
        x=(1+cos(i))*cos(i-pi/2)
        y=(1+cos(i))*sin(i-pi/2)+2
        x_list.append(x)
        y_list.append(y)


    turtle1=Turtle()
    turtle2=Turtle()
    turtle1.color('black','red')
    turtle2.color('black','red')

    turtle2.begin_fill()
    for i in range(len(x_list)):
        x=x_list[i]*100
        y=y_list[i]*100
        turtle1.setpos(x,y)
        y = y_list[i] * 98
        turtle2.setpos(x,y)
    turtle2.end_fill()
    done()

drawHeart()
