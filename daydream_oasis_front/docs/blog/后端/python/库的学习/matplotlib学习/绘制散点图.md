import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0,10,100)

#用scatter绘制散点图

plt.show()

"""
从上面的绘图情况可以知道使用plot和scatter都可以绘制散点图，但，plot的绘制速度比scatter快，
