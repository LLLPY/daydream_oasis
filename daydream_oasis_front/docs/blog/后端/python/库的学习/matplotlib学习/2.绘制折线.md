```python
#导入matplotlib.pylot模块
import matplotlib.pyplot as plt

#将多个不在同一条直线上的点连在一起构成折线
import numpy as np
x = np.arange(1,13)
y = np.power(x,2)

#绘制图像:连接各点
plt.plot(x,y)

#显示图像 y = x**2
plt.show()
```
