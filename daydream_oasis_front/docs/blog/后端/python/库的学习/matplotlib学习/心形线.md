```python
import matplotlib.pyplot as plt
from math import sin,cos,pi
import numpy as np

x_list=[]
y_list=[]
r=1

for t in np.arange(0,2*pi,0.1):
    x=r*(1-cos(t))*cos(t)
    y=r*(1-cos(t))*sin(t)
    x_list.append(x)
    y_list.append(y)


plt.plot(x_list,y_list)
plt.show()
```
