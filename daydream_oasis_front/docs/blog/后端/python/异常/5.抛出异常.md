```python
def input_password():
    # 1.提示用户输入密码
    password = input("请输入密码：")

    # 2.判断密码长度是否>=8
    if len(password) >= 8:
        return password

    # 3.主动抛出异常
    # print("主动抛出异常")

    # 1>创建一个异常对象
    error = ValueError("密码长度不够")

    # 2>使用raise关键字将异常对象抛出
    raise error


# 捕获异常

pwd = input_password()
print(pwd)
```
