```python
hello_str = "hello world!!!"

#统计字符串的长度
print(len(hello_str))

#统计某个小字符串出现的次数
print(hello_str.count("h"))

#判断某一个子字符串出现的位置
print(hello_str.index("h"))
print(hello_str.index("he"))
```
