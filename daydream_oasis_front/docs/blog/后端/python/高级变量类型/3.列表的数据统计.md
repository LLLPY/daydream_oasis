```python
name_list = ["张三","李四","王五","张三"]

#len（length长度）函数  可以统计列表中元素的总和
length = len(name_list)
print("列表中有%d个元素"%length)
#count 方法  可以统计列表中某一个数据出现的次数
times = name_list.count("张三")
print("张三在列表中出现了%d次"%times)
name_list.remove("张三")
#若要删除的数据在列表中重复出现，remove会删除
#出现的第一个数据
print(name_list)
# reverse :翻转
```
