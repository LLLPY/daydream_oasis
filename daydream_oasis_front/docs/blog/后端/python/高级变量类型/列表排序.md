```python
name_list = ["张三","李四","王五"]
num_list = [122,33,45,67,45,25,89,90]
#升序排列
name_list.sort()
num_list.sort()
print(num_list)
print(name_list)
#翻转排列（逆序排列）
name_list.reverse()
num_list.reverse()
print(name_list)
print(num_list)
#降序排列
name_list.sort(reverse=True)
print(name_list)
num_list.sort(reverse=True)
print(num_list)


```
