```python
#提示：在日常开发中，要从列表中删除数据，建议使用列表提供的方法
name_list = ["zhangsan","lisi","wangwu"]
#使用del关键字（delete）删除列表元素
del name_list[0]
# del关键字本质上是将一个变量从内存中删除
print(name_list)
name = "小明"
del name
 #注意：如果使用del关键字将变量从内存中删除
 #后续代码就不能再使用这个变量了

```
