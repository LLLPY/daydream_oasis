```python
#假设：以下内容是从网络上抓取的
#要求：顺序并且居中对齐输出以下内容
poem = ["登鹳雀楼",
        "王之涣",
        "白日依山尽",
        "黄河入海流",
        "欲穷千里路",
        "更上一层楼"]

#居左打印
for poem_str in poem:
    print("|%s|"%poem_str.ljust(12," "))
#居中打印
for poem_str1 in poem:
    print("|%s|"%poem_str1.center(12," "))
#居右打印
for poem_str2 in poem:
    print("|%s|"%poem_str2.rjust(12," "))
```
