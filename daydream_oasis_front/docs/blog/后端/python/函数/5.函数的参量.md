```python
#其中num1和num2是形参，是形式上的参数
def sum_2_num(num1,num2):
    result = num1 + num2
    print("%d + %d = %d"%(num1,num2,result))
#10和23是实参，是实实在在的参数，程序运行时，将实参的值传递给
#实参，且按照函数定义的参数顺序依次传递
sum_2_num(10,23)
#使用函数的参量，增加了函数的通用性，针对相同的数据处理
#逻辑，能够适应更多的数据，增强了函数的功能
```
