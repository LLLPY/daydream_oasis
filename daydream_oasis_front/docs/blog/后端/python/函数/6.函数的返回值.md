#函数的返回值
#关键字：return
def sum_2_num(num1,num2):
    #对任意两个数字求和
    result = num1 + num2
    #定义result变量用来存计算的结果
    return result
    print("hello world!!!")
    #注意：return就表示返回，后面的代码不会再执行
    #将计算的结果返回

sum_result = sum_2_num(12,23)
#定义sun_result变量用来接收返回值
print("结果是：%d "%sum_result)
