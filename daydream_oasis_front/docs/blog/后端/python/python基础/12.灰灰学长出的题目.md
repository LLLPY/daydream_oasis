#要求：去除数组中的重复数
#数组为：111223443355544
A = [1,1,1,2,2,3,4,4,3,3,5,5,5,4,4]
row = 1
while row<=5:
    print("")
    col = 1
    while col<=row:
        print("*",end="")
        col += 1
    row += 1

print(list(set(A)))
