```python
#定义变量记录数据
i = 0
# 定义变量记录计算结果
result = 0
#设定循环次数
while i <= 100 :
    #处理计数器
    i += 1
#筛选数据，选出0到100之间的偶数
    if i % 2 == 0 :
        result += i
print("0到100之间的计算结果是 ：%d" % result)
```
