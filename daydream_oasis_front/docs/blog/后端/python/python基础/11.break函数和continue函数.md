```python
#break函数
#在循环语句中，若输入break关键字，会直接打断循环，不再执行循环的代码
i = 0
while i < 10 :
    print("Hello python")
    if i == 3 : # i = 3时，也执行代码
         break
    i += 1
#continue函数
i = 0
while i < 10 :
    if i == 3 :
        print("hello word")
        i += 1
        continue
    i += 1
    print("hello python")
```
