```python
def funcout(func):
    def funcin(*args):
        return func(*args) #将闭包函数的内涵数的参数改为*args，则闭包函数就可以接受任意多个参数的函数，提高了其兼容性
    return funcin

@funcout
def add(a,b,c,d):
    return a+b+c+d

@funcout
def sum(a,b):
    return a+b

print('10+20+30+40=',add(10,20,30,40))
print('10+20=',sum(10,20))
```
