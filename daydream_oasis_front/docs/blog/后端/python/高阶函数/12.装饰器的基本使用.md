import time

def writelog(func):
    try:
        f = open('log.txt', 'a', encoding='utf-8')
        f.write('功能:')
        f.write(func.__name__)
        f.write('\t')
        f.write('时间:')
        f.write(time.asctime())
        f.write('\n')
    except Exception as result:
        print(result.args)
    finally:
        f.close()

#闭包函数
def funcout(func):
    def funcin():
        writelog(func)
        func()
    return funcin

@funcout
def func1():
    print('功能1')



def func2():
    print('功能2')

func1()



#将闭包的操作简化
#在需要调用的函数上一行用修饰器修饰闭包函数
#然后在调用原函数(闭包函数的内函数)
