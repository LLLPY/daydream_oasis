```python
import math

def funcout(x,y):
    def funcin(a,b):
        return math.sqrt(pow(x-a,2) + pow(y-b,2))
    return funcin

#求(1,2),(3,5)两点之间的距离
length = funcout(1,2)(3,5)
print('两点之间的距离为:',length)
```
