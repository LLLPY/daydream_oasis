```python
import numpy

#sorted是高阶函数，它还可以接受一个key函数来实现自定义的排序

a = list(numpy.random.randint(-100,100,(20)))

#按照绝对值的大小进行排序
print(a)
b = sorted(a,key=abs)
print(b)


```
