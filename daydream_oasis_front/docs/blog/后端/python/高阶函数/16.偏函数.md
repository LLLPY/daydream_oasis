```python
a = '10'

print(int(a)) #int()函数默认将a转化成10进制的整数 base=10:说明原整数是一个10进制数
print(int(a,base=2))

print(int('1010',base=2)) #base=2是为了说明1010是一个二进制的数

#偏函数(固定函数的参数)
from functools import partial
new_int2 = partial(int,base=2)

def new_int(a):
    return int(a,base=2)



print(new_int('1010'))
print(new_int2('1010'))
```
