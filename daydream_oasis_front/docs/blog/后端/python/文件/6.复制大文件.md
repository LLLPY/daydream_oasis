```python
#以只读方式打开
file_read = open("lll02sayhello")

#以只写方式打开
file_write = open("[复件]","w")

while True:
    # 读
    text = file_read.read()
    #判断是否读到内容，如果没有，则突出循环
    if not text:
        break
    #写
    file_write.write(text)



#关闭文件
file_read.close()
file_write.close()
```
