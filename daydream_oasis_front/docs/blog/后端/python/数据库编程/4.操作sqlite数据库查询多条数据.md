```python
#导入模块
import sqlite3

#创建连接对象
con=sqlite3.connect('C:/Users/LLL/Desktop/python/python基础(演练)/数据库编程/SQLite3数据库/demo1.db')

#创建游标对象
cur=con.cursor()

#编写sql查询语句
sql='select * from t_person'

try:
    #游标对象执行sql查询语句
    cur.execute(sql)
    person_all=cur.fetchall() #fetchall()查询所有数据
    print(person_all)
except Exception as r:
    print(r)
    print('查询失败!')
finally:
    #关闭游标对象
    cur.close()
    #关闭数据库连接
    con.close()
```
