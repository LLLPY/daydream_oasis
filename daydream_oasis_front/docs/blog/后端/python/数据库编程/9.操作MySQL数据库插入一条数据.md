```python
import pymysql

#创建连接对象
con =pymysql.connect(host='127.0.0.1',port=3306,user='root',password='123456',database='python_db')

#创建游标对象
cur=con.cursor()

#编写插入的sql语句
sql='insert into t_student(sname,age,score) values(%s,%s,%s)'

try:
    #执行插入的sql语句
    cur.execute(sql,('小明',18,99))
    #提交事务
    con.commit()
    print('插入数据成功!')
except Exception as e:
    print(e)
    #事务回滚
    con.rollback()
    print('数据插入失败!')
finally:
    #关闭数据库的连接
    con.close()

```
