```python
import sqlite3

con=sqlite3.connect('db.sqlite3')
cur=con.cursor()
sql='update t_person set pname=? where pno=?'
try:
    cur.execute(sql,('小张',1))
    con.commit()
    print('修改数据成功!')
except Exception as r:
    print(r)
    print('修改数据失败!')
finally:
    cur.close()
    con.close()
```
