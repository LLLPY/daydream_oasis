    <script>
        //语法:
        /* 
         do {
             循环体
         }while(循环条件)
         
          */

        //dowhile循环先执行一次循环体，不管循环条件是否满足，所以do-while循环无论如何都至少执行一次

        //代码体验
        do {
            var str = prompt('你爱我吗?');
        } while (str != '我爱你')
        alert('可惜我不爱你，哈哈!!!');
    </script>
