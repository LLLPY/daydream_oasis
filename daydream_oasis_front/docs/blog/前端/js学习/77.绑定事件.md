```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>绑定事件</title>
</head>
<script>

    window.onload = function () {

        /*
        * 事件绑定分为两大类
        *
        * 1.内联模式
        *
        * 2.外联模式/脚本模式(使用较多)
        *
        * 绑定事件的格式:
        * 元素节点.on+事件类型(例:click)=匿名函数.
        *
        * click:类型
        * onclick:事件处理的函数
        *
        * */

        var out = document.getElementById('out');
        out.onclick = function () {
            alert('外联模式!!!');
        }
        inclick = function () {
            alert('内联模式!!!');
        }

    }

</script>
<body>

<button onclick="inclick();">内联模式
</button>
<button id="out">外联模式</button>

</body>
</html>
```
