```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>遍历对象</title>0;
    <script>
        //对象是一个无序的集合，所以简单的循环遍历不适用，但是可以用for in 进行遍历

        var obj = new Object();
        obj.name = '小明';
        obj.age = 18;
        obj.weight = 70;
        obj.height = 178;
        obj.hobby = '打游戏';

        for (var k in obj) {
            console.log(k, obj[k]);
        }
    </script>
</head>

<body>

</body>
```
