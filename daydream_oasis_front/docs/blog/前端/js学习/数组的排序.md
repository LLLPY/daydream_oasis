    <script>
        var arr = [1, 34, 23, 3, 2, 2, 4, 6, 67, 45, -56, 45, 34 - 3434];
        //数组的翻转
        console.log(arr.reverse());
        //数组的排序
        //1.不传参数，默认方法排序
        console.log(arr.sort());
        //2.升序排序
        arr.sort(function(a, b) {
            return a - b;
        })
        console.log(arr);
        //3.降序排序
        arr.sort(function(a, b) {
            return b - a;
        })
        console.log(arr);
    </script>
