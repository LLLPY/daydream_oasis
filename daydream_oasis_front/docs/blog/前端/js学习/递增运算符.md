<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>递增运算符</title>
    <script>
        var num = 1;

        //进行加一运算
        num = num + 1;
        console.log(num);
        //简便写法:运用递增运算符

        ++num; //前置递增运算符 先自加，后返回值
        console.log(num);
        //前置运算符和后置运算符在单独使用计算时产生的效果一模一样

        num++; //后置递增运算符 先返回值，后自加
        console.log(num);

        var num2 = num3 = 10;
        console.log(num2++ + 1); //11
        console.log(num2); //11

        console.log(++num3 + 1); //12
        console.log(num3); //11
    </script>
</head>
