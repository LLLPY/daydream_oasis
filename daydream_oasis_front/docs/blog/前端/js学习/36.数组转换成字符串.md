<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>数组转换车成字符串</title>
    <script>
        var arr = [1, 3, 4, 243, ];
        //1.toString()方法进行转换
        console.log(arr.toString());
        //2.join(连接符)方法进行转换(不传参数，默认连接符是逗号)
        console.log(arr.join());
        console.log(arr.join('-')); //用-进行连接各个元素
    </script>
</head>
