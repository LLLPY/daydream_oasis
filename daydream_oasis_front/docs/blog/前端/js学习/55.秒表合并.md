```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>秒表案例</title>
    <style>
        * {
            padding: 0;
            margin: 0;
            outline: 0;
        }

        #div1 {

            width: 200px;
            height: 300px;
            margin: 100px auto;
            background-color: greenyellow;
            border: 2px solid yellow;
            border-radius: 15px;
            text-align: center;

        }

        button {
            width: 70%;
            height: 50px;
            background-color: aqua;
            margin-top: 10px;
            border-radius: 10px;


        }

        button:hover {
            background-color: limegreen;
        }

        button:active {
            background-color: red;
        }

        #start {
            margin-top: 50px;
        }

        span {
            display: inline-block;
            width: 50px;
            height: 50px;
            background-color: green;
            margin-top: 20px;
            text-align: center;
            line-height: 50px;
            font-size: 30px;
            border-radius: 10px;
        }


    </style>
    <script>

        /*简化通过id获取元素的方法*/
        function $(id) { /*变量命名可用$*/
            return document.getElementById(id);
        }

        window.onload = function () {
            var i = 0;
            var timer = null;
            var isRuning = false;

            //返回两位数
            function reDoble(num) {
                return num >= 10 ? num : '0' + num;
            }

            function startFunc() {
                //    开始计时
                timer = setInterval(function () {
                    i++;

                    var se = i % 60;
                    var mi = parseInt(i / 60) % 60;
                    var ho = parseInt(i / 3600);

                    $("second").innerHTML = reDoble(se);
                    $("minute").innerHTML = reDoble(mi);
                    $("hour").innerHTML = reDoble(ho);
                }, 1000);

            }

            //获取暂停按钮
            function pauseFunc() {
                clearInterval(timer);
            }

            //给按钮添加点击事件
            $('sta-pa').onclick = function () {
                if (!isRuning) {
                    $("sta-pa").innerHTML = "暂停";
                    startFunc();
                    isRuning = true;
                } else {
                    pauseFunc();
                    $("sta-pa").innerHTML = "开始";
                    isRuning = false;
                }


            }


            //复位
            $('restart').onclick = function () {
                /*首先清理掉计时器*/
                clearInterval(timer);
                i = 0;
                $("second").innerHTML = '00';
                $("minute").innerHTML = '00';
                $("hour").innerHTML = '00';
                $("sta-pa").innerHTML = "开始";
                isRuning = false;


            }


        }
    </script>
</head>

<body>

<div id="div1">


    <span id="hour">00</span>
    <span id="minute">00</span>
    <span id="second">00</span>
    <button id="sta-pa">开始</button>
    <button id="restart">复位</button>

</div>

</body>
</html>
```
