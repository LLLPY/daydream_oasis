```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>数字补0案例</title>
    <script>
        //输入一个0~59之间的数字，如果数字小于10则在数字前面补一个0
        var num = prompt('请输入一个0~59的数字:');
        while (num < 0 || num > 59) {
            alert('您的输入有误,请重新输入!!!')
            var num = prompt('请输入一个0~59的数字:');
        }

        //ifelse的写法
        if (num <= 9) {
            alert('0' + num);
        } else {
            alert(num);
        }

        //三元表达式的写法
        var result = num <= 9 ? '0' + num : num;
        alert(result);
    </script>
</head>

<body>

</body>

</html>
```
