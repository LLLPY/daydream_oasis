<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>判断闰年的案例</title>
    <script>
        //判断是否是闰年:如果年份能被4整除且不能被100整除或者能被400整除，则该年份是闰年
        var year = prompt('请输入一个年份:');
        if (year % 4 == 0 && year % 100 != 0 || year % 400 == 0) {
            alert(year + '是闰年!');
        } else {
            alert(year + '不是闰年!');
        }
    </script>
</head>

</body>
