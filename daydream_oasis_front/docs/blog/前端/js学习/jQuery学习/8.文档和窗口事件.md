```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>文档/窗口事件</title>
    <script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>

    <script>

        /*
        * 文档/窗口事件
        *
        * 1.resize 当调整浏览器窗口大小时,触发该事件
        * 2.scroll 该事件适用于所有可滚动的元素和窗口对象
        *
        *
        *
        * */

        /*文档预加载*/

        $(
            function () {

                var html_tag = $('html');

                html_tag.resize(
                    function () {
                        alert('窗口大小发生了改变');
                    }
                );


                /*滚动窗口*/
                html_tag.scroll(
                    function () {
                        alert('滚动了窗口');
                    }
                );


            }
        );


    </script>


</head>
<body style="height: 2000px;">

</body>
</html>
```
