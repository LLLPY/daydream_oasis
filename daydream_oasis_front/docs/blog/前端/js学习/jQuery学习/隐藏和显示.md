```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>隐藏和显示</title>
    <script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>

    <script>
        /*
        * 隐藏和显示
        *
        *1.hide(speed,func,callback);
        *2.show(speed,func,callback);
        *3.toggle(speed,func,callback)
        *
        *可选的speed参数规定隐藏/显示的速度,可以取以下值:slow,fast或者毫秒
        *可选的callback参数是隐藏或显示完成后要执行的函数的函数名称
        *可选的func参数表示用哪种缓冲函数
        *
        * */

        $(
            function () {
                var div_tag=$('div');
                var hide_tag=$('button:first-child');
                var show_tag=$('button:nth-child(2)');
                var show_hide_tag=$('button:nth-child(3)');
                hide_tag.click(
                    function () {
                    div_tag.hide(500);
                    }
                );
                show_tag.click(
                    function () {
                    div_tag.show(500);
                    }
                );
                 show_hide_tag.click(
                    function () {
                    div_tag.toggle();
                    }
                );
            }
        );
    </script>


</head>
<body>

<button>隐藏</button>
<button>显示</button>

<button>显示/隐藏</button>

<div>hello world</div>
```
