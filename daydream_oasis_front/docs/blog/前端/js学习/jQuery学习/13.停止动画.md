```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>停止动画</title>
    <script src="https://cdn.bootcss.com/jquery/1.10.2/jquery.min.js"></script>

    <script>
        /*
        *
        * stop()方法用于停止动画或效果,在它们完成之前
        * stop(stopAll,goToEnd)方法适用于所有jquery效果函数,包括滑动,淡入淡出和自定义动画
        *可选参数stopAll规定是否应该清除动画队列.默认是false,即仅停止活动的动画,允许任何排入
        * 队列的动画向后执行
        * 可选参数goToEnd参数规定是否立即完成当前动画,默认是false
        * 因此,默认地,stop()会清除在被选元素上指定的动画
        *
        * */

        $(function () {
            $("#box1").click(
                function () {
                    $("#box2").slideToggle(2000);
                }
            );


            /*点击停止动画*/
            $("#stop").click(function () {
                $("#box2").stop(true); /*如果设置为true,会停止当前元素的所有动画*/
            });


        });

        setTimeout(function () {

            $("#ex").animate({"width": "+=200px"}, 1000);


        }, 3000);


    </script>

    <style>
        div {
            width: 200px;
            height: 100px;
            margin: 0 auto;
            text-align: center;
            background-color: greenyellow;
            line-height: 100px;
            border: 1px solid red;
        }

        div:last-child {
            width: 80%;
            margin-top: 5px;
            font-size: 30px;
        }

        #stop {
            width: 50px;
            height: 50px;
            border: 1px solid red;
            float: right;
            text-align: center;
            line-height: 50px;
            background-color: red;
            margin-right: 100px;


        }

        #ex {
            width: 100px;
            height: 100px;
            background-color: rebeccapurple;
        }


    </style>


</head>
<body>
<div id="ex"></div>
<span id="stop">停止</span>

<div id="box1">显示/隐藏菜单</div>
<div id="box2">happy middle autumn festival!</div>


</body>
</html>
```
