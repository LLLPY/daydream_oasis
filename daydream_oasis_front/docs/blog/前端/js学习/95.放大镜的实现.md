    * {

    }

    #small {

    #big {
        width: 400px;
        height: 400px;
        position: absolute;
        left: 550px;

        border: 2px solid black;
        overflow: hidden;
    }


    #p1 {
        width: 100%;

    }

    #p2 {
        position: absolute;
        left: 0;

        display: none;
    }

    #mark {
        width: 100px;
        height: 100px;
        background-color: white;
        opacity: 0.5; /*背景透明*/

<script>

    window.onload = function () {

        var small = document.getElementById('small');
        var p2 = document.getElementById('p2');
        var omark = document.getElementById('mark');

        //鼠标移入,mark显示出来
        small.onmouseover = function () {
            omark.style.display = 'block';
            p2.style.display = 'block';

        }

        small.onmouseout = function () {
            omark.style.display = 'none';
            p2.style.display = 'none';

        }

        //鼠标移入small后,mark跟随鼠标移动,且mark不能出界
        small.onmousemove = function (ev) {
            var e = ev || window.event;

            //计算small距离窗口左上角的距离
            var small_l = small.offsetLeft;
            var small_t = small.offsetTop;
            var l = e.clientX - small_l - 50;
            var t = e.clientY - small_t - 50;
            //不允许出界
            if (l <= 0) {
                l = 0;
            }
            if (l >= 380) {
                l = 380;
            }
            if (t <= 0) {
                t = 0;
            }
            if (t >= 170) {
                t = 170;
            }

            //右边图片移动的方式:左边遮盖层如何移动,右边的图片,反方向对应倍数移动
            p2.style.left = l * -4 + 'px'; //这里是放大4x4倍
            p2.style.top = t * -4 + 'px';

            omark.style.left = l + 'px';
            omark.style.top = t + 'px';

        }

    }


</script>
