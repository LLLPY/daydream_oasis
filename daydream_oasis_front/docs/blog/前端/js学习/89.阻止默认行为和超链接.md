```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>阻止默认行为和超链接</title>
</head>
<script>
    //默认情况下,在页面上,鼠标点击右键会弹出一个菜单(这是官方的右键菜单)
    window.onload = function () {
        //取消官方的右键菜单
        document.oncontextmenu = function () {
            return false;
        }

        //自定义右键菜单
        var menus = document.getElementById('menu');
        document.onmousedown = function (ev) {
            var e = ev || window.event;
            if (e.button === 2) {  //如果点击的是右键
                menus.style.display = 'block';
                menus.style.left = e.clientX + 'px';
                menus.style.top = e.clientY + 'px';
            } else {
                menus.style.display = 'none';
            }

        }

        //默认情况下,在网页上点击一个链接,会自动跳转到对应的网页
        /*
        * 阻止超链接的几种方法
        * 1.直接return false
        * */

        //找到a链接 为a添加点击事件
        var a = document.getElementById('aa');
        a.onclick = function (ev) {
            preDef(ev);
            // return false; //return false后不会跳转
            return confirm('您确定跳转到新的页面吗?') //跳转前询问一下
        }

        //方法2(使用的官方提供的方法)
        //阻止超链接的浏览器兼容写法 (这个方法的好处在于,可以把把阻止的代码放在前面,且后续的代码
        // 仍会执行,如果使用return 则不行)
        function preDef(ev) {
            if (ev.preventDefault) {
                ev.preventDefault();
            } else {
                window.event.returnValue = false;
            }

        }

    }


</script>

<style>
    div {
        display: none;
        background-color: aqua;
        position: absolute;
    }
</style>
<body>
<a href="https://www.baidu.com" id="aa">百度</a>

<div id="menu">
    <li><a href="#">菜单1</a></li>
    <li><a href="#">菜单2</a></li>
    <li><a href="#">菜单3</a></li>


</div>
</body>
</html>
```
