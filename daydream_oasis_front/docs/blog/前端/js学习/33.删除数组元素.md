<!DOCTYPE html>

    <title>删除数组元素</title>
    <script>
        var arr = [1, 2, 3, 4, 5, 6];
        //1.pop() 删除数组的最后一个元素
        arr.pop();
        console.log(arr);
        //pop()完毕后返回的是删除的元素
        console.log(arr.pop());

        //2.shift() 删除数组的第一个元素
        arr.shift();
        console.log(arr);
        //同样，shift()完毕后返回的删除的元素
        console.log(arr.shift());
    </script>
</head>
