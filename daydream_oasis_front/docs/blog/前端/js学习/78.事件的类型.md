```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>事件的类型</title>
</head>
<script>

    /*
    * 事件类型的种类:
    * 一鼠标事件:
    *   click:单击
    *   dblclick:双击
    *   mouseover:鼠标移入
    *   mouseout:鼠标移出
    *
    *   mousemove:鼠标移动(会不停的触发)
    *   mouseenter:鼠标移入
    *   mouseleave:鼠标移出
    *   mouseup:鼠标抬起
    *
    * 二.
    *   键盘事件
    *
    * 三.HTML事件
    *   1.window事件
    *
    *   2.表单事件
    *
    *
    *
    *
    *
    *
    * */

    window.onclick=function () {

        var bu=document.getElementById('bu');
        var odiv=document.getElementById('odiv');
        bu.onclick=function () {
            odiv.innerHTML='你单击了按钮!!!';
        }
        bu.ondblclick=function () {
            odiv.innerHTML='你双击了按钮!!!';

        }
        odiv.onmouseover=function () {
                        odiv.innerHTML='鼠标移入!!!';
                odiv.style.backgroundColor='red';
        }
        odiv.onmouseout=function () {
                        odiv.innerHTML='鼠标移出!!!';
                odiv.style.backgroundColor='green';
        }
        bu.onmouseenter=function () {
            bu.innerHTML='鼠标移入!!!';
                bu.style.backgroundColor='red';

        }
         bu.onmouseout=function () {
            bu.innerHTML='鼠标移出!!!';
                bu.style.backgroundColor='yellow';

        }

        bu.onmousedown=function () {
            bu.innerHTML='鼠标按下!!!';
                bu.style.backgroundColor='blue';

        }

         bu.onmouseup=function () {
            bu.innerHTML='鼠标抬起!!!';
                bu.style.backgroundColor='white';
        }
        var i=0;
        odiv.onmousemove=function () {
            i+=1;
            odiv.innerHTML=i;
        }


    }


</script>
<style>
    #bu{
        width: 100px;
        height: 100px;
    }
    #odiv{
        width: 200px;
        height: 200px;
        text-align: center;
        background-color: chartreuse;
        line-height: 200px;
    }

</style>

<body>
<button id="bu">按钮</button>
<div id="odiv">啦啦啦~~</div>
</body>
</html>
```
