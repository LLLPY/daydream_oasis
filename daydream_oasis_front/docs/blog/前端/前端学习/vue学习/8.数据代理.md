```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>数据代理</title>
    <script src="js/vue.js"></script>
</head>
<body>
<script>

    /*
    *
    * 数据代理：通过一个对象代理对另外一个对象中属性的操作(读/写)
    *
    * Vue中数据代理的原理：
    *   通过Object.defineProperty()把data对象中所有属性添加到vm上。
    *   为每一个添加到vm上的属性都指定一个getter和一个setter
    *   在getter/setter内部去操作data中对应的属性
    *
    * */

    //让obj2代理obj1的属性x
    let obj1 = {x: 100}
    let obj2 = {}

    Object.defineProperty(obj2, 'x', {
        enumerable:true,

        get() {
            return obj1.x
        },
        set(v) {
            obj1.x = v
        }
    })

</script>


</body>
</html>
```
