```html
<!doctype html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport"
          content="width=device-width, user-scalable=no, initial-scale=1.0, maximum-scale=1.0, minimum-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>类选择器</title>
    <style>
        h3 { color : red;}
        .apple {color:green;} <!-- 类选择器的声明-->
    </style>
</head>
<body>
<h3 class="apple">苹果</h3> <!--类选择器的定义-->
<h3>香蕉</h3>
<h3>白菜</h3>
</body>
</html>
```
