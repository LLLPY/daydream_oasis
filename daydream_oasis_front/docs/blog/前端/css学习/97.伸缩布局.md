```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>伸缩布局</title>
    <style>
        nav {
            width:900px;
            height:300px;
            border: 2px solid red;
            margin:100px auto;
            display:flex;
            flex-direction:column;  /*调整主轴方向为垂直方向(column),默认是水平的(row)*/
            }

        div {

            height:300px;
            float:left;
            background-color:purple;
            border:2px solid skyblue;
            flex:1;      /*中间的盒子占1/5(1+2+2)*/
            margin-left:5px;
            margin-right:5px;

            }
       div:nth-child(2n-1) {
            background-color:green;
            flex:2;    /*两边的盒子各占2/5*/
            margin:0;

            }


    </style>
</head>
<body>


<nav>
    <div></div>
    <div></div>
    <div></div>
</nav>

</body>
```
